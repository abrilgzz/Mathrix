matrix int m[1][1];
matrix int n[1][1];
matrix int r[1][1];
int i;
int j;
int k;
int t;

Main {
    i = 0;
    j = 0;
    k = 0;

    matrix m = {[1,2],[3,4]};
    matrix n = {[1,1],[1,1]};
    matrix r = {[0,0],[0,0]};


    while(i < 2){
        while(j < 2){
            r[i][j] = 0;
            while(k < 2){
                t = r[i][j];
                r[i][j] = t + m[i][k] * n[k][j];
                k = k + 1;
            }
            j = j + 1;
        }
        i = i + 1;
    }

    write(r[0][0]);
    write(r[0][1]);
    write(r[1][0]);
    write(r[1][1]);
}
